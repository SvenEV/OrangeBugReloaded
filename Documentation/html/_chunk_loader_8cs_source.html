102  public async Task<IChunk> TryGetAsync(Point index)
103  {
104  // If chunk already loaded, return it immediately
105  IChunk alreadyLoadedChunk;
106  if (Chunks.TryGetValue(index, out alreadyLoadedChunk))
107  return alreadyLoadedChunk;
108 
109  // If chunk is already being loaded, await its loader task (do not start a second task)
110  LoadingChunkToken loadingChunkToken;
111  if (ChunksLoading.TryGetValue(index, out loadingChunkToken))
112  return await loadingChunkToken.LoaderTask;
113 
114  // Otherwise, begin loading the chunk
115  var token = new LoadingChunkToken { Cancellation = new CancellationTokenSource() };
116  ChunksLoading.Add(index, token);
117 
118  IChunk chunk = null;
119 
120  try
121  {
122  token.LoaderTask = Storage.LoadAsync(index, token.Cancellation.Token);
123  chunk = await token.LoaderTask;
124  token.Cancellation.Token.ThrowIfCancellationRequested(); // Handle cancellation in case storage doesn't
125  }
126  catch (OperationCanceledException)
127  {
128  // Chunk loading was cancelled by an Unload(...) call, do nothing
129  }
130  catch
131  {
132  // Something else happened, do nothing
133  }
134 
135  if (chunk == null)
136  {
137  // If chunk loading fails, create an empty chunk
138  // (empty chunks won't be saved)
139  chunk = new Chunk(index);
140  }
141  else
142  {
143  // The storage might have its own caching strategies and
144  // reuse chunks it has once loaded, so we need to work with
145  // a copy of the loaded chunk.
146  chunk = (IChunk)chunk.Clone();
147  }
148 
149  ChunksLoading.Remove(index);
150  Chunks.Add(index, chunk);
151  return chunk;
152  }
