94  public async Task<bool> MoveAsync(Point sourcePosition, Point targetPosition, EntityMoveTransaction transaction)
95  {
96  // Execute the actual move.
97  // Any changes to the map are "recorded" in the transaction
98  // and can later be applied to the map via a commit.
99  await MoveCoreAsync(sourcePosition, targetPosition, transaction);
100 
101  if (transaction.IsCancelled)
102  return false;
103 
104  // At the end of the recursion...
105  if (transaction.Moves.Count == 0)
106  {
107  var notifiedPositions = new HashSet<Point>();
108 
109  // Tiles that definitely must be notified are the ones that changed
110  // and the ones that depend on any of the changed tiles.
111  var affectedTiles = transaction.ChangedTiles.Select(kvp => kvp.Key)
112  .Concat(transaction.ChangedTiles.SelectMany(p => Dependencies.GetDependenciesOn(p.Key)));
113 
114  // Notify affected tiles and dependent tiles in topological order
115  await Dependencies.DoAsyncWorkFollowingDependenciesAsync(affectedTiles, async p =>
116  {
117  notifiedPositions.Add(p);
118  var tile = await transaction.Last.GetAsync(p);
119  var completionArgs = new TileEventArgs(transaction);
120  await tile.OnEntityMoveTransactionCompletedAsync(completionArgs);
121  completionArgs.ValidateResult();
122 
123  if (transaction.IsCancelled)
124  return null; // Null terminates the loop
125 
126  return await transaction.SetAsync(p, completionArgs.Result);
127  });
128 
129  // Trigger follow-up transactions
130  // e.g. a teleporter might initiate a new transaction here to teleport an entity
131  var args = new FollowUpEventArgs(transaction);
132 
133  foreach (var p in notifiedPositions)
134  {
135  var tile = await transaction.Last.GetAsync(p);
136  await tile.OnFollowUpTransactionAsync(args, p);
137  }
138  }
139 
140  return true;
141  }

143  private async Task MoveCoreAsync(Point sourcePosition, Point targetPosition, EntityMoveTransaction transaction)
144  {
145  var move = new EntityMoveInfo
146  {
147  SourcePosition = sourcePosition,
148  TargetPosition = targetPosition
149  };
150 
151  transaction.Moves.Push(move);
152 
153  var source = await transaction.GetAsync(sourcePosition);
154  var target = await transaction.GetAsync(targetPosition);
155  move.Entity = source.Entity;
156 
157  // OnBeginMove: Notify entity that a move has been initiated
158  var beginMoveArgs = new EntityEventArgs(transaction);
159  await source.Entity.BeginMoveAsync(beginMoveArgs);
160  beginMoveArgs.ValidateResult();
161  if (transaction.IsCancelled) return;
162  source = Tile.Compose(source, beginMoveArgs.Result);
163  await transaction.SetAsync(sourcePosition, source);
164  move.Entity = beginMoveArgs.Result;
165 
166  // Detach: Remove the entity from the source tile
167  var detachArgs = new TileEventArgs(transaction);
168  await source.DetachEntityAsync(detachArgs);
169  detachArgs.ValidateResult();
170  if (transaction.IsCancelled) return;
171  source = detachArgs.Result;
172  await transaction.SetAsync(sourcePosition, source);
173 
174  // Attach: Add the entity to the target tile
175  var attachArgs = new TileEventArgs(transaction);
176  await target.AttachEntityAsync(attachArgs);
177  attachArgs.ValidateResult();
178  if (transaction.IsCancelled) return;
179  target = attachArgs.Result;
180  await transaction.SetAsync(targetPosition, target);
181 
182  transaction.Moves.Pop();
183  }
