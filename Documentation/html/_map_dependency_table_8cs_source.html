162  public async Task DoAsyncWorkFollowingDependenciesAsync(IEnumerable<Point> initialBag, Func<Point, Task<bool?>> action)
163  {
164  var bag = new HashSet<Point>(initialBag);
165  var counter = 0;
166 
167  while (bag.Any())
168  {
169  var current = bag.FirstOrDefault(p => _dependencies.TryGetValue(p, _emptySet).All(q => !bag.Contains(q)));
170 
171  // Note: This does not fully protect us from cycles, there could still be
172  // cases where P and Q go alternately in and out of the bag
173  if (current == null || counter++ > 1000000)
174  throw new InvalidOperationException("Dependency cycle detected");
175 
176  var hasChanged = await action(current);
177 
178  if (!hasChanged.HasValue)
179  return; // Null has been returned -> terminate algorithm
180 
181  // Remove visited point from bag
182  bag.Remove(current);
183 
184  if (hasChanged.Value)
185  {
186  // Add all points that depend on the visited point
187  foreach (var dependentPoint in _dependentOn.TryGetValue(current, _emptySet))
188  bag.Add(dependentPoint);
189  }
190  }
